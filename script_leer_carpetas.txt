# Ruta de la carpeta src
$srcPath = "D:\Ano_2025_UCR\Analisis\Proyecto\frontend_proton_react_hogar_de_ancianos\src"

# Función recursiva para mostrar carpetas y archivos
function Show-Tree {
    param(
        [string]$Path,
        [int]$Level = 0
    )

    $indent = " " * ($Level * 4)

    # Mostrar carpetas
    Get-ChildItem -Path $Path -Directory | ForEach-Object {
        Write-Output "$indent📁 $_"
        # Llamada recursiva para subcarpetas
        Show-Tree -Path $_.FullName -Level ($Level + 1)
    }

    # Mostrar archivos
    Get-ChildItem -Path $Path -File | ForEach-Object {
        Write-Output "$indent- $_"
    }
}

# Ejecutar función
Show-Tree -Path $srcPath
